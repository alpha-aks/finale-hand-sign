import os

    url += "hand_landmarker/hand_landmarker/float16/1/"
    url += "hand_landmarker.task"
    urllib.request.urlretrieve(url, model_path)

# Configure for dual-hand detection

    base_options=base_options,
    num_hands=2,  # Changed from 1 to 2

)

\subsection{Gesture Classifier Implementation}

\begin{lstlisting}[language=Python]
class KeyPointClassifier:
    def __init__(self, model_path='model/keypoint_classifier/'
                           'keypoint_classifier.tflite'):
        self.interpreter = tf.lite.Interpreter(
            model_path=model_path)
        self.interpreter.allocate_tensors()
        
        self.input_details = self.interpreter.get_input_details()
        self.output_details = self.interpreter.get_output_details()

    def __call__(self, landmark_list):
        input_data = np.array(
            [landmark_list], dtype=np.float32)
        
        self.interpreter.set_tensor(
            self.input_details[0]['index'], input_data)
        self.interpreter.invoke()
        
        output_data = self.interpreter.get_tensor(
            self.output_details[0]['index'])
        
        return np.argmax(output_data[0])
\end{lstlisting}

\subsection{Text-to-Speech Integration}

\begin{lstlisting}[language=Python]
def speak_text(text):
    """Speak text in background thread"""
    def _speak():
        try:
            engine = pyttsx3.init()
            engine.setProperty('rate', 150)
            engine.say(text)
            engine.runAndWait()
            engine.stop()
        except Exception:
            pass
    
    thread = threading.Thread(target=_speak, daemon=True)
    thread.start()

# Usage in typing mode
if st.button("🔊 Speak"):
    if st.session_state.typed_text:
        speak_text(st.session_state.typed_text)
\end{lstlisting}

\section{Model Training}

\subsection{Training Process}

\begin{lstlisting}[language=Python]
# Load and preprocess data
X_train, y_train = load_training_data('training_data/')

# Normalize labels
y_train_categorical = tf.keras.utils.to_categorical(
    y_train, num_classes=num_gestures)

# Build model
model = tf.keras.Sequential([
    tf.keras.layers.Input(shape=(42,)),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dropout(0.2),
    tf.keras.layers.Dense(64, activation='relu'),
    tf.keras.layers.Dropout(0.2),
    tf.keras.layers.Dense(num_gestures, activation='softmax')
])

# Compile and train
model.compile(
    optimizer='adam',
    loss='categorical_crossentropy',
    metrics=['accuracy']
)

history = model.fit(
    X_train, y_train_categorical,
    epochs=50,
    batch_size=32,
    validation_split=0.2,
    verbose=1
)

# Save models
model.save('model/keypoint_classifier/keypoint_classifier.hdf5')
converter = tf.lite.TFLiteConverter.from_keras_model(model)
tflite_model = converter.convert()
with open('model/keypoint_classifier/keypoint_classifier.tflite',
          'wb') as f:
    f.write(tflite_model)
\end{lstlisting}

\section{Performance Optimizations}

\subsection{Real-Time Processing}

\begin{enumerate}
    \item \textbf{GPU Acceleration}: TensorFlow uses GPU if available
    \item \textbf{Model Quantization}: TFLite models are quantized for speed
    \item \textbf{Threading}: Non-blocking operations (speech synthesis)
    \item \textbf{Caching}: Streamlit caches MediaPipe models
    \item \textbf{Efficient Preprocessing}: Vectorized NumPy operations
\end{enumerate}

\subsection{Memory Management}

\begin{itemize}
    \item Deque with fixed maximum length for point history
    \item Image buffering in video capture
    \item Session state for Streamlit persistence
\end{itemize}
